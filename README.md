## Install

```bash
npm install --save lwh-ui-second
```

## Usage

```jsx
import React, { Component } from 'react'

import {Header,Footer} from 'lwh-ui-second'

class Example extends Component {
  render () {
    return (
      <Header />
      <Footer />
    )
  }
}
```
